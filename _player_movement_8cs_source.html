2// ARCHIVO: PlayerMovement.cs
3// PROPÓSITO: Controlar el movimiento completo del jugador:
4// desplazamiento horizontal, salto, agachado, volteo de sprite,
5// detección de suelo/paredes y sistema de knockback.
6// RESPONSABILIDAD: Gestionar el input de movimiento y traducirlo en
7// física y animaciones de forma coherente con el estado de combate.
8// SISTEMA: Movimiento — Física y Control del Jugador

11/*
12 * DESCRIPCIÓN DEL SISTEMA
13 * -----------------------
14 * Este script es el núcleo del control de movimiento del jugador.
15 * Procesa el input de teclado para mover, saltar y agacharse,
16 * ajustando el Collider2D dinámicamente al agacharse y detectando
17 * colisiones con suelo, techo y paredes mediante raycasts y OverlapCircle.
18 *
19 * Implementa un sistema de knockback (ApplyKnockback) que deshabilita
20 * temporalmente el script completo para que el movimiento no interfiera
21 * con la física del golpe. Al terminar el knockback, el control se restaura.
22 *
23 * También detecta colisiones con otros personajes (characterLayer) para
24 * evitar que los jugadores se empujen entre sí durante el movimiento normal.
25 *
26 * SISTEMA: Movimiento / Física del Jugador
27 * INTERACTÚA CON: PlayerAttack.cs (isAttacking), Rigidbody2D, Animator
28 */
29

244 /// <summary>
245 /// Start() — Se ejecuta una vez al iniciar.
246 /// Verifica que boxCollider esté asignado (requerido), guarda las
247 /// dimensiones originales del collider y calcula las del agachado.
248 /// También obtiene el Rigidbody2D si no fue asignado en el Inspector.
249 ///
250 /// NOTA: Si boxCollider es null, el script se deshabilita automáticamente
251 /// para evitar NullReferenceExceptions en Update/FixedUpdate.
252 /// </summary>
253 void Start()
254 {
255 // Guard: boxCollider es necesario antes de leer su tamaño
256 if (boxCollider == null)
257 {
258 Debug.LogError("boxCollider no está asignado en el Inspector de " + gameObject.name, this);
259 enabled = false;
260 return;
261 }
262
263 // Guardar dimensiones originales para restaurar al salir del agachado
264 originalSize = boxCollider.size;
265 originalOffset = boxCollider.offset;
266 originalScale = transform.localScale;
267
268 // Calcular dimensiones del collider en agachado (mitad de altura)
269 crouchSize = new Vector2(originalSize.x, originalSize.y / 2);
270 crouchOffset = new Vector2(originalOffset.x, originalOffset.y - (originalSize.y / 4));
271
272 // Obtener Rigidbody2D si no fue asignado en el Inspector
273 if (rb == null)
274 rb = GetComponent<Rigidbody2D>();
275 }
276
277 /// <summary>
278 /// Update() — Se ejecuta cada frame.
279 /// Procesa todo el input de movimiento, salto, agachado y volteo de sprite.
280 /// Actualiza el estado del Animator y redimensiona el collider según el estado.
281 ///
282 /// CUÁNDO SE EJECUTA: Cada frame. Si isKnockedBack = true, retorna inmediatamente.
283 /// QUÉ CONTROLA:
284 /// - Input horizontal → moveInput
285 /// - Detección de paredes y agachado
286 /// - Salto con verificación de suelo
287 /// - Volteo del sprite (Flip)
288 /// - Parámetros del Animator (Speed, isJumping, isCrouching)
289 /// - Redimensionado del BoxCollider2D al agacharse/levantarse
290 /// </summary>
291 void Update()
292 {
293 // Bloquear todo input durante el vuelo del knockback
294 if (isKnockedBack) return;
295
296 // Guard: verificar referencias críticas antes de ejecutar lógica
297 if (wallCheck == null || groundCheck == null || ceilingCheck == null || boxCollider == null || animator == null)
298 {
299 Debug.LogError("Faltan referencias requeridas en el Inspector de " + gameObject.name, this);
300 return;
301 }
302
303 float input = 0;
304
305 if (Input.GetKey(leftKey))
306 input = -1;
307 else if (Input.GetKey(rightKey))
308 input = 1;
309
310 // Detectar pared lateral con raycast en la dirección del input
311 if (input != 0)
312 isTouchingWall = Physics2D.Raycast(wallCheck.position, Vector2.right * Mathf.Sign(input), wallCheckDistance, groundLayer);
313 else
314 isTouchingWall = false;
315
316 // Determinar velocidad efectiva según estado del personaje
317 if (isTouchingWall)
318 moveInput = 0;
319 else if (isCrouching)
320 // Guard: evitar división por cero si speed es inválido
321 moveInput = speed > 0 ? input * (crouchSpeed / speed) : 0;
322 else
323 moveInput = input;
324
325 animator.SetFloat("Speed", Mathf.Abs(moveInput));
326
327 // Voltear el sprite si cambia la dirección horizontal
328 if (moveInput > 0 && !facingRight)
329 Flip();
330 else if (moveInput < 0 && facingRight)
331 Flip();
332
333 // Detectar si el personaje toca el suelo
334 isGrounded = Physics2D.OverlapCircle(groundCheck.position, 0.2f, groundLayer);
335
336 // Detectar techo para bloquear levantarse del agachado
337 isBlocked = Physics2D.OverlapCircle(ceilingCheck.position, 0.2f, groundLayer);
338
339 if (Input.GetKeyDown(jumpKey) && isGrounded && !isCrouching)
340 rb.linearVelocity = new Vector2(rb.linearVelocity.x, jumpForce);
341
342 animator.SetBool("isJumping", !isGrounded);
343
344 if (Input.GetKey(crouchKey))
345 isCrouching = true;
346 else if (!isBlocked)
347 isCrouching = false;
348
349 animator.SetBool("isCrouching", isCrouching);
350
351 // Redimensionar el BoxCollider2D al agacharse y al levantarse
352 if (isCrouching)
353 {
354 boxCollider.size = crouchSize;
355 boxCollider.offset = crouchOffset;
356 }
357 else
358 {
359 boxCollider.size = originalSize;
360 boxCollider.offset = originalOffset;
361 }
362 }
363
364 /// <summary>
365 /// FixedUpdate() — Se ejecuta en el ciclo de física (por defecto 50Hz).
366 /// Aplica la velocidad horizontal calculada en Update() al Rigidbody2D.
367 /// También detecta colisiones con otros personajes y detiene el movimiento
368 /// al hacer contacto para evitar empujes físicos entre jugadores.
369 ///
370 /// CUÁNDO SE EJECUTA: Cada paso de física. Si isKnockedBack = true, retorna.
371 /// QUÉ CONTROLA:
372 /// - Aplicación de velocidad horizontal al Rigidbody2D.
373 /// - Detección de otros personajes mediante RaycastAll con characterLayer.
374 /// - Bloqueo de movimiento al tocar a otro personaje.
375 /// </summary>
376 void FixedUpdate()
377 {
378 if (isKnockedBack) return;
379
380 // wallCheck es requerido antes de ejecutar el raycast
381 if (wallCheck == null) return;
382
383 float move = moveInput;
384
385 if (moveInput != 0)
386 {
387 // RaycastAll en characterLayer para detectar colisión con otro personaje
388 RaycastHit2D[] hits = Physics2D.RaycastAll(
389 wallCheck.position,
390 Vector2.right * Mathf.Sign(moveInput),
391 0.1f,
392 characterLayer);
393
394 foreach (RaycastHit2D hit in hits)
395 {
396 // Saltar colliders que pertenecen a esta misma jerarquía (autocolisión)
397 if (hit.collider.transform.root == transform.root) continue;
398
399 bool hitsCharacter = ((1 << hit.collider.gameObject.layer) & characterLayer) != 0;
400
401 if (hitsCharacter)
402 {
403 // Detener este personaje al hacer contacto con el oponente
404 move = 0;
405 rb.linearVelocity = new Vector2(0, rb.linearVelocity.y);
406
407 // No zerear velocidad del oponente si está en knockback activo
408 Rigidbody2D otherRb = hit.collider.transform.root.GetComponent<Rigidbody2D>();
409 if (otherRb != null)
410 {
411 PlayerMovement otherMovement = hit.collider.transform.root.GetComponent<PlayerMovement>();
412 if (otherMovement == null || !otherMovement.isKnockedBack)
413 otherRb.linearVelocity = new Vector2(0, otherRb.linearVelocity.y);
414 }
415
416 break;
417 }
418 }
419 }
420
421 rb.linearVelocity = new Vector2(move * speed, rb.linearVelocity.y);
422 }

428 /// <summary>
429 /// Invierte la dirección del sprite horizontalmente.
430 /// Cambia facingRight y niega la escala X del transform.
431 ///
432 /// CUÁNDO SE LLAMA: Desde Update() cuando moveInput cambia de signo.
433 /// QUÉ AFECTA: transform.localScale.x y el flag facingRight.
434 /// </summary>
435 void Flip()
436 {
437 facingRight = !facingRight;
438
439 // Invertir escala horizontal para voltear el sprite
440 Vector3 scale = transform.localScale;
441 scale.x *= -1;
442 transform.localScale = scale;
443 }

445 // ============================================================
446 // API PÚBLICA — KNOCKBACK
447 // ============================================================

449 /// <summary>
450 /// Aplica un knockback al jugador desde una posición de atacante.
451 /// Deshabilita el script completo durante knockbackDuration para que
452 /// el input no interfiera con el vuelo del knockback.
453 ///
454 /// CUÁNDO SE LLAMA: Externamente cuando el jugador recibe un golpe
455 /// (no usado actualmente en el flujo principal; ver HealthSystem).
456 /// QUÉ AFECTA: isKnockedBack, rb (fuerza), this.enabled (script deshabilitado).
457 ///
458 /// NOTA: canBeKnockedBack e isKnockedBack previenen knockback doble.
459 /// </summary>
460 /// <param name="attackerPosition">Posición del atacante para calcular dirección del vuelo.</param>
461 public void ApplyKnockback(Vector2 attackerPosition)
462 {
463 if (!canBeKnockedBack || isKnockedBack) return;
464
465 isKnockedBack = true;
466
467 // Calcular dirección opuesta al atacante
468 Vector2 direction = (transform.position - (Vector3)attackerPosition).normalized;
469
470 if (rb != null)
471 {
472 rb.linearVelocity = Vector2.zero;
473 // El eje Y tiene un valor fijo de 2f para pequeña elevación del vuelo
474 rb.AddForce(new Vector2(direction.x * knockbackForce, 2f), ForceMode2D.Impulse);
475 }
476
477 // Deshabilitar script para que Update no interfiera durante el knockback
478 this.enabled = false;
479
480 // Mathf.Max evita que duration negativo rompa el Invoke
481 Invoke(nameof(EndKnockback), Mathf.Max(0f, knockbackDuration));
482 }
483
484 /// <summary>
485 /// Finaliza el knockback y restaura el control del jugador.
486 /// Se llama automáticamente mediante Invoke() al vencer knockbackDuration.
487 /// Limpia la velocidad horizontal residual al aterrizar.
488 /// </summary>
489 private void EndKnockback()
490 {
491 isKnockedBack = false;
492
493 // Restaurar control del jugador al terminar el knockback
494 this.enabled = true;
495
496 // Eliminar velocidad horizontal residual para aterrizar limpiamente
497 if (rb != null)
498 rb.linearVelocity = new Vector2(0, rb.linearVelocity.y);
499 }
500
501 // ============================================================
502 // FLUJO DEL SISTEMA
503 // ============================================================
504 /*
505 * MOVIMIENTO NORMAL:
506 * 1. Update() lee input → calcula moveInput según estado (agachado, pared, libre).
507 * 2. Update() voltea sprite si cambia dirección, detecta suelo/techo, actualiza Animator.
508 * 3. Update() redimensiona BoxCollider2D si cambia estado de agachado.
509 * 4. FixedUpdate() aplica moveInput * speed al Rigidbody2D.
510 * 5. FixedUpdate() detecta otros personajes → bloquea movimiento al contacto.
511 *
512 * KNOCKBACK:
513 * 1. ApplyKnockback(position) → isKnockedBack = true, this.enabled = false.
514 * 2. Update/FixedUpdate retornan inmediatamente → sin input.
515 * 3. Rigidbody2D vuela libremente por la fuerza de knockback.
516 * 4. Después de knockbackDuration → EndKnockback() → this.enabled = true.
517 * 5. Velocidad horizontal residual limpiada → jugador recupera control.
518 */
519

524 * ES AFECTADO POR:
525 * - HealthSystem.cs → deshabilita/habilita este script durante hit-stun.

535 // ============================================================
536 // NOTAS / ADVERTENCIAS
537 // ============================================================
538 /*
539 * ⚠ boxCollider ES OBLIGATORIO. Sin él, el script se deshabilita en Start()
540 * y el jugador quedará completamente inmovilizado.
541 *
542 * ⚠ characterLayer DEBE incluir tanto la capa "Player" como "Duplicate"
543 * (u otras capas de personajes). Si falta alguna capa, los personajes
544 * se empujarán físicamente en lugar de detenerse.
545 *
546 * ⚠ ApplyKnockback() en este script no es el sistema principal de knockback
547 * para el jugador. HealthSystem.cs gestiona el knockback del jugador
548 * deshabilitando directamente este script. ApplyKnockback() está disponible
549 * como alternativa pero actualmente no se llama en el flujo de combate.
550 *
551 * ⚠ Invoke() no se cancela si el script se deshabilita.
552 * EndKnockback() se ejecutará aunque el objeto esté desactivado.
553 * Agregar CancelInvoke(nameof(EndKnockback)) en OnDisable() si esto
554 * pudiera causar problemas en escenas con respawn.
555 *
556 * ⚠ La detección de personajes en FixedUpdate usa RaycastAll con distancia 0.1f.
557 * Si los personajes están muy separados o el frame rate es muy bajo,
558 * pueden "atravesarse" antes de que el raycast los detecte.
559 */
