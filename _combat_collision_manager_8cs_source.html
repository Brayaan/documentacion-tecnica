1// ============================================================
2// ARCHIVO: CombatCollisionManager.cs
3// PROPÓSITO: Gestionar las colisiones físicas entre enemigos
4// en el sistema de combate del juego.
5// RESPONSABILIDAD: Configurar la física de los enemigos al iniciar
6// la escena para evitar que se empujen entre sí.
7// SISTEMA: Combate — Física y Colisiones
8// ============================================================
9
10/*
11 * DESCRIPCIÓN DEL SISTEMA
12 * -----------------------
13 * Este script actúa como inicializador global del comportamiento físico
14 * de todos los enemigos en la escena. Resuelve el problema de que los
15 * enemigos se empujen mutuamente al acumularse cerca del jugador,
16 * lo que rompe la lógica de combate.
17 *
18 * Se ejecuta una sola vez en Start() y configura las colisiones
19 * entre todos los GameObjects con el tag "Enemy" usando Physics2D.IgnoreCollision.
20 *
21 * SISTEMA: Combate / Físicas
22 * DEPENDE DE: Collider2D y Rigidbody2D en cada enemigo
23 */
24
25using UnityEngine;
26
27public class CombatCollisionManager : MonoBehaviour
28{
29 // -------------------------
30 // VARIABLES PÚBLICAS
31 // -------------------------
32
33 /// <summary>
34 /// Si es true, los enemigos podrán empujarse físicamente entre sí.
35 /// Si es false (por defecto), sus colisiones físicas se ignorarán.
36 /// Configurable desde el Inspector para ajustar el diseño de gameplay.
37 /// </summary>
38 public bool enemiesPushEachOther = false;
39
40 // ============================================================
41 // EVENTOS DE UNITY
42 // ============================================================
43
44 /// <summary>
45 /// Start() — Se ejecuta una sola vez al iniciar la escena.
46 /// Dispara la configuración global de física para todos los enemigos.
47 /// Es el punto de entrada único de este sistema.
48 /// </summary>
49 void Start()
50 {
51 ConfigureAllEnemies();
52 }
53
54 // ============================================================
55 // MÉTODOS PRINCIPALES
56 // ============================================================
57
58 /// <summary>
59 /// Busca todos los GameObjects con el tag "Enemy" en la escena
60 /// y les aplica la configuración de físicas correspondiente.
61 ///
62 /// CUÁNDO SE EJECUTA: Una sola vez al inicio, invocado desde Start().
63 /// QUÉ AFECTA: Todos los enemigos activos en la escena al momento de Start().
64 ///
65 /// NOTA: Los enemigos instanciados después de Start() NO serán configurados.
66 /// Si el juego usa spawn dinámico, llamar ConfigureEnemyPhysics()
67 /// manualmente al instanciar cada enemigo.
68 /// </summary>
69 private void ConfigureAllEnemies()
70 {
71 // FindGameObjectsWithTag lanza excepción si el tag no existe
72 GameObject[] enemies;
73 try
74 {
75 enemies = GameObject.FindGameObjectsWithTag("Enemy");
76 }
77 catch (UnityEngine.UnityException)
78 {
79 Debug.LogError("El tag 'Enemy' no está registrado en Project Settings > Tags", this);
80 return;
81 }
82
83 // Obtener todos los colliders una sola vez antes del loop
84 // para evitar llamadas repetidas costosas a FindObjectsByType dentro del bucle
85 Collider2D[] allColliders = FindObjectsByType<Collider2D>(FindObjectsSortMode.None);
86
87 foreach (GameObject enemy in enemies)
88 {
89 ConfigureEnemyPhysics(enemy, allColliders);
90 }
91 }
92
93 /// <summary>
94 /// Configura las físicas de un enemigo individual:
95 /// - Bloquea la rotación del Rigidbody2D.
96 /// - Hace que este enemigo ignore colisiones físicas con todos los demás enemigos.
97 ///
98 /// QUÉ HACE: Itera sobre todos los Collider2D de la escena
99 /// y aplica Physics2D.IgnoreCollision entre enemigos.
100 /// CUÁNDO SE EJECUTA: Una vez por enemigo, llamado desde ConfigureAllEnemies().
101 /// QUÉ AFECTA: Rigidbody2D y Collider2D del enemigo recibido como parámetro.
102 /// </summary>
103 /// <param name="enemy">El GameObject del enemigo a configurar.</param>
104 /// <param name="allColliders">Lista de todos los Collider2D activos en la escena.</param>
105 private void ConfigureEnemyPhysics(GameObject enemy, Collider2D[] allColliders)
106 {
107 Rigidbody2D rb = enemy.GetComponent<Rigidbody2D>();
108
109 if (rb != null)
110 {
111 if (!enemiesPushEachOther)
112 {
113 // Prevenir que la física rote al Rigidbody
114 rb.constraints = RigidbodyConstraints2D.FreezeRotation;
115
116 Collider2D myCollider = enemy.GetComponent<Collider2D>();
117
118 // Saltar enemigos sin Collider2D para evitar NullReferenceException
119 if (myCollider == null)
120 {
121 Debug.LogWarning("El enemigo " + enemy.name + " no tiene Collider2D", enemy);
122 return;
123 }
124
125 // Ignorar colisiones físicas entre todos los enemigos de la escena
126 foreach (Collider2D other in allColliders)
127 {
128 if (other.gameObject.CompareTag("Enemy") && other.gameObject != enemy)
129 {
130 Physics2D.IgnoreCollision(myCollider, other, true);
131 }
132 }
133 }
134 }
135 }
136
137 // ============================================================
138 // FLUJO DEL SISTEMA
139 // ============================================================
140 /*
141 * 1. Start() → ConfigureAllEnemies()
142 * 2. Se obtienen todos los Enemy tags y todos los Collider2D de la escena.
143 * 3. Por cada enemigo → ConfigureEnemyPhysics()
144 * 4. Se congela la rotación del Rigidbody2D del enemigo.
145 * 5. Se llama Physics2D.IgnoreCollision entre el enemigo y cada otro enemigo.
146 * 6. El sistema queda configurado estáticamente para toda la partida.
147 */
148
149 // ============================================================
150 // INTERACCIONES CON OTROS SISTEMAS
151 // ============================================================
152 /*
153 * DEPENDE DE:
154 * - Rigidbody2D en cada enemigo (requerido para configurar constraints).
155 * - Collider2D en cada enemigo (requerido para IgnoreCollision).
156 * - Tag "Enemy" registrado en Project Settings > Tags.
157 *
158 * AFECTA A:
159 * - EnemyHealthSystem: al eliminar el empuje entre enemigos,
160 * el knockback recibido no se ve interferido por colisiones de otros enemigos.
161 *
162 * NO ES AFECTADO POR ningún otro script en runtime.
163 */
164
165 // ============================================================
166 // NOTAS / ADVERTENCIAS
167 // ============================================================
168 /*
169 * ⚠ ADVERTENCIA: Solo configura enemigos presentes en escena al momento de Start().
170 * Los enemigos generados dinámicamente deben configurarse manualmente.
171 *
172 * ⚠ Si se agrega un nuevo tag para variantes de enemigos (ej: "BossEnemy"),
173 * este script no los cubrirá automáticamente. Extender ConfigureAllEnemies()
174 * para incluir los tags adicionales.
175 *
176 * ⚠ NO modificar enemiesPushEachOther en runtime; el sistema no reaplica
177 * la configuración después de Start().
178 */
179}
