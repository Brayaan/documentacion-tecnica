1// ============================================================
2// ARCHIVO: NoPushOnHit.cs
3// PROPÓSITO: Evitar que el jugador sea desplazado físicamente
4// por colisiones con enemigos durante el hit-stun.
5// RESPONSABILIDAD: Aumentar temporalmente la masa del Rigidbody2D
6// para resistir empujes físicos al recibir un golpe.
7// SISTEMA: Combate — Física de Impacto / Hit-Stun
8// ============================================================
9
10/*
11 * DESCRIPCIÓN DEL SISTEMA
12 * -----------------------
13 * Cuando el jugador recibe un golpe y entra en hit-stun, otros enemigos
14 * que lo toquen físicamente podrían empujarlo accidentalmente, acumulando
15 * desplazamientos no deseados. Este script resuelve ese problema aumentando
16 * la masa del Rigidbody2D a un valor extremo (1000f) durante el hit-stun,
17 * lo que hace que las fuerzas de contacto sean prácticamente inefectivas.
18 *
19 * Al terminar el hit-stun, la masa se restaura al valor original.
20 * También ignora colisiones físicas con enemigos mientras isHit está activo.
21 *
22 * SISTEMA: Combate / Física
23 * INTERACTÚA CON: HealthSystem.cs (debe llamar OnHitStart/OnHitEnd manualmente)
24 *
25 * NOTA: Este script expone OnHitStart() y OnHitEnd() como API pública.
26 * Deben ser llamados externamente por el sistema que gestiona el hit-stun.
27 */
28
29using UnityEngine;
30
31public class NoPushOnHit : MonoBehaviour
32{
33 // -------------------------
34 // VARIABLES PRIVADAS
35 // -------------------------
36
37 /// <summary>
38 /// Referencia al Rigidbody2D del jugador.
39 /// Se obtiene en Start(). Necesario para modificar la masa y las constraints.
40 /// </summary>
41 private Rigidbody2D rb;
42
43 /// <summary>
44 /// Masa original del Rigidbody2D antes de entrar en hit-stun.
45 /// Se guarda en Start() y se restaura en OnHitEnd() para no perder
46 /// la configuración del Inspector.
47 /// </summary>
48 private float originalMass;
49
50 /// <summary>
51 /// Flag de estado de hit-stun activo.
52 /// true = el jugador está en hit-stun; colisiones con enemigos se ignoran.
53 /// false = comportamiento físico normal.
54 /// </summary>
55 private bool isHit = false;
56
57 // ============================================================
58 // EVENTOS DE UNITY
59 // ============================================================
60
61 /// <summary>
62 /// Start() — Se ejecuta una vez al iniciar.
63 /// Obtiene el Rigidbody2D y guarda la masa original como referencia.
64 /// </summary>
65 void Start()
66 {
67 rb = GetComponent<Rigidbody2D>();
68 if (rb != null)
69 {
70 // Guardar masa original para restaurarla al terminar el golpe
71 originalMass = rb.mass;
72 }
73 }
74
75 /// <summary>
76 /// OnCollisionEnter2D — Se ejecuta al entrar en colisión con otro Collider2D
77 /// (colisión sólida, no trigger).
78 ///
79 /// CUÁNDO SE EJECUTA: Al primer frame de contacto físico con otro collider.
80 /// QUÉ CONTROLA: Durante el hit-stun, ignora colisiones físicas con enemigos
81 /// para evitar que lo empujen mientras ya está volando por el knockback.
82 ///
83 /// NOTA: Physics2D.IgnoreCollision persiste entre activaciones del objeto.
84 /// Si el enemigo cambia de estado, la colisión permanece ignorada
85 /// hasta que se reactive explícitamente.
86 /// </summary>
87 void OnCollisionEnter2D(Collision2D collision)
88 {
89 // Durante el hit-stun: ignorar colisiones físicas con cualquier Enemy
90 if (isHit && collision.gameObject.CompareTag("Enemy"))
91 {
92 Physics2D.IgnoreCollision(GetComponent<Collider2D>(), collision.collider, true);
93 }
94 }
95
96 // ============================================================
97 // MÉTODOS PRINCIPALES — API PÚBLICA
98 // ============================================================
99
100 /// <summary>
101 /// Inicia el estado de hit-stun para este personaje.
102 /// Aumenta la masa del Rigidbody2D a 1000f para resistir empujes físicos
103 /// de contacto con otros colliders, y congela la rotación.
104 ///
105 /// CUÁNDO SE LLAMA: Debe ser llamado externamente por el sistema de hit-stun
106 /// (HealthSystem.cs o similar) al inicio de un golpe recibido.
107 /// QUÉ AFECTA: rb.mass (modificado a 1000f), rb.constraints, flag isHit.
108 ///
109 /// ADVERTENCIA: Si OnHitEnd() nunca se llama, el jugador quedará con
110 /// masa de 1000f permanentemente, rompiendo su física.
111 /// </summary>
112 public void OnHitStart()
113 {
114 if (rb != null)
115 {
116 // Masa extrema para resistir desplazamientos por contacto físico
117 rb.mass = 1000f;
118 rb.constraints = RigidbodyConstraints2D.FreezeRotation;
119 }
120 isHit = true;
121 }
122
123 /// <summary>
124 /// Finaliza el estado de hit-stun y restaura la física al estado normal.
125 /// Restaura la masa original guardada en Start().
126 ///
127 /// CUÁNDO SE LLAMA: Debe ser llamado externamente al terminar el hit-stun.
128 /// Típicamente mediante Invoke() o una coroutine desde el
129 /// script que llamó OnHitStart().
130 /// QUÉ AFECTA: rb.mass (restaurado), flag isHit.
131 /// </summary>
132 public void OnHitEnd()
133 {
134 if (rb != null)
135 {
136 // Restaurar masa original al salir del hit-stun
137 rb.mass = originalMass;
138 }
139 isHit = false;
140 }
141
142 // ============================================================
143 // FLUJO DEL SISTEMA
144 // ============================================================
145 /*
146 * 1. Start() → guarda originalMass del Rigidbody2D.
147 * 2. HealthSystem (o sistema externo) detecta un golpe → llama OnHitStart().
148 * 3. OnHitStart() → rb.mass = 1000f, isHit = true.
149 * 4. Durante el hit-stun: si un enemigo colisiona → OnCollisionEnter2D()
150 * → Physics2D.IgnoreCollision activo para ese enemigo.
151 * 5. Al terminar el hit-stun → OnHitEnd() → rb.mass = originalMass, isHit = false.
152 */
153
154 // ============================================================
155 // INTERACCIONES CON OTROS SISTEMAS
156 // ============================================================
157 /*
158 * ES LLAMADO POR:
159 * - HealthSystem.cs (o el sistema que gestione el hit-stun del jugador)
160 * debe llamar OnHitStart() y OnHitEnd() manualmente.
161 *
162 * AFECTA A:
163 * - Rigidbody2D del jugador (masa y constraints).
164 * - Sistema de colisiones de Physics2D (IgnoreCollision con enemigos).
165 *
166 * NOTA IMPORTANTE: En la versión actual del proyecto, HealthSystem.cs
167 * NO llama a OnHitStart/OnHitEnd. Este script está preparado como
168 * extensión pero requiere integración manual para funcionar.
169 */
170
171 // ============================================================
172 // NOTAS / ADVERTENCIAS
173 // ============================================================
174 /*
175 * ⚠ CRÍTICO: OnHitStart() y OnHitEnd() deben llamarse en par.
176 * Si OnHitEnd() no se llama (por error o excepción), el jugador
177 * quedará con rb.mass = 1000f permanentemente, haciendo que la
178 * física de movimiento y salto sea incorrecta.
179 *
180 * ⚠ Physics2D.IgnoreCollision() en OnCollisionEnter2D persiste
181 * indefinidamente. Si el mismo enemigo vuelve a colisionar después
182 * del hit-stun, la colisión seguirá ignorada hasta que se reactive
183 * con Physics2D.IgnoreCollision(col1, col2, false).
184 *
185 * ⚠ Este script no gestiona el timer del hit-stun.
186 * El control de duración debe hacerse desde el script que lo llama.
187 *
188 * ⚠ La masa de 1000f es un valor hardcodeado. Si la masa original del
189 * Rigidbody2D es muy alta (ej: 500f), la diferencia puede no ser
190 * suficiente. Considerar usar rb.isKinematic = true en su lugar.
191 */
192}
