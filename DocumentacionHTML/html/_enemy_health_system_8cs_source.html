1// ============================================================
2// ARCHIVO: EnemyHealthSystem.cs
3// PROPÓSITO: Gestionar la vida, el daño recibido, el knockback
4// y la muerte de un enemigo individual.
5// RESPONSABILIDAD: Punto de entrada de daño para todos los enemigos.
6// SISTEMA: Combate — Salud y Reacción al Daño (Enemigos)
7// ============================================================
8
9/*
10 * DESCRIPCIÓN DEL SISTEMA
11 * -----------------------
12 * Este script controla la salud de cada enemigo de forma independiente.
13 * Cuando el jugador golpea a un enemigo, llama a TakeDamage() para
14 * reducir su vida, aplicar un knockback físico y verificar si debe morir.
15 *
16 * Implementa un sistema de hit-stun (isHit) que previene que golpes
17 * muy rápidos apilen fuerzas de knockback simultáneas, manteniendo
18 * el comportamiento físico predecible y justo.
19 *
20 * SISTEMA: Combate / Salud de Enemigos
21 * INTERACTÚA CON: Hitbox.cs, ExtendedHitbox.cs
22 */
23
24using UnityEngine;
25
26public class EnemyHealthSystem : MonoBehaviour
27{
28 // -------------------------
29 // VARIABLES PÚBLICAS
30 // -------------------------
31
32 /// <summary>
33 /// Vida máxima del enemigo. Define cuántos puntos de daño puede absorber
34 /// antes de morir. Configurable por tipo de enemigo desde el Inspector.
35 /// </summary>
36 public int maxHealth = 50;
37
38 /// <summary>
39 /// Vida actual del enemigo en tiempo real.
40 /// Se inicializa igual a maxHealth en Start().
41 /// Visible en el Inspector para debug durante la partida.
42 /// </summary>
43 public int currentHealth;
44
45 /// <summary>
46 /// Fuerza del empuje físico (knockback) que recibe el enemigo al ser golpeado.
47 /// Configurable desde el Inspector. Valor más alto = mayor desplazamiento.
48 /// </summary>
49 public float knockbackForce = 5f;
50
51 /// <summary>
52 /// Duración en segundos del hit-stun: tiempo mínimo que debe pasar
53 /// antes de que el enemigo pueda recibir otro knockback.
54 /// Evita que golpes rápidos apilen fuerzas y lo lancen fuera de la pantalla.
55 /// </summary>
56 public float knockbackDuration = 0.3f;
57
58 // -------------------------
59 // VARIABLES PRIVADAS
60 // -------------------------
61
62 /// <summary>
63 /// Referencia al Rigidbody2D del enemigo.
64 /// Se obtiene automáticamente en Start(). Necesario para aplicar knockback.
65 /// </summary>
66 private Rigidbody2D rb;
67
68 /// <summary>
69 /// Flag de hit-stun activo.
70 /// true = el enemigo está en hit-stun y no puede recibir otro knockback.
71 /// false = el enemigo puede recibir daño con efecto físico completo.
72 /// Previene el "stacking" de fuerzas por golpes consecutivos rápidos.
73 /// </summary>
74 private bool isHit = false;
75
76 // ============================================================
77 // EVENTOS DE UNITY
78 // ============================================================
79
80 /// <summary>
81 /// Start() — Se ejecuta una vez al activarse el enemigo en escena.
82 /// Inicializa la vida al máximo y obtiene el Rigidbody2D.
83 /// </summary>
84 void Start()
85 {
86 currentHealth = maxHealth;
87 rb = GetComponent<Rigidbody2D>();
88 }
89
90 // ============================================================
91 // MÉTODOS PRINCIPALES
92 // ============================================================
93
94 /// <summary>
95 /// Punto de entrada de daño para el enemigo.
96 /// Reduce la vida, aplica knockback y verifica si el enemigo debe morir.
97 ///
98 /// CUÁNDO SE LLAMA: Desde Hitbox.cs o ExtendedHitbox.cs cuando un golpe conecta.
99 /// QUÉ AFECTA: currentHealth, física del Rigidbody2D, y destrucción del objeto.
100 /// </summary>
101 /// <param name="damage">Cantidad de puntos de vida a restar.</param>
102 /// <param name="attackerPosition">Posición del atacante para calcular dirección del knockback.</param>
103 public void TakeDamage(int damage, Vector2 attackerPosition)
104 {
105 currentHealth -= damage;
106
107 // Clampear vida para no bajar de cero (evita valores negativos en UI o lógica)
108 if (currentHealth < 0)
109 currentHealth = 0;
110
111 Debug.Log($"Enemigo vida: {currentHealth}");
112
113 ApplyHit(attackerPosition);
114
115 // Verificar muerte después de aplicar el golpe para permitir animación de hit
116 if (currentHealth <= 0)
117 Die();
118 }
119
120 // ============================================================
121 // MÉTODOS AUXILIARES
122 // ============================================================
123
124 /// <summary>
125 /// Aplica el knockback físico al enemigo cuando recibe un golpe.
126 /// Controla el hit-stun para evitar que fuerzas se acumulen.
127 ///
128 /// CUÁNDO SE EJECUTA: Llamado internamente desde TakeDamage().
129 /// QUÉ AFECTA: Rigidbody2D (velocidad y fuerza), flag isHit.
130 ///
131 /// NOTA: Si isHit ya está activo, el método retorna sin hacer nada.
132 /// Esto es intencional para evitar stacking de knockback.
133 /// </summary>
134 /// <param name="attackerPosition">Posición desde donde vino el ataque.</param>
135 void ApplyHit(Vector2 attackerPosition)
136 {
137 // Ignorar golpe si el hit-stun aún está activo
138 if (isHit) return;
139
140 isHit = true;
141
142 // Dirección del empuje: opuesta al atacante para alejarse de él
143 Vector2 direction = (transform.position - (Vector3)attackerPosition).normalized;
144
145 if (rb != null)
146 {
147 rb.linearVelocity = Vector2.zero;
148 // El eje Y tiene un valor fijo de 2f para dar una pequeña elevación al golpe
149 rb.AddForce(new Vector2(direction.x * knockbackForce, 2f), ForceMode2D.Impulse);
150 }
151
152 // Mathf.Max evita que un knockbackDuration negativo rompa el Invoke
153 Invoke(nameof(EndHit), Mathf.Max(0f, knockbackDuration));
154 }
155
156 /// <summary>
157 /// Finaliza el hit-stun y permite que el enemigo reciba knockback nuevamente.
158 /// Se llama automáticamente mediante Invoke() al vencer knockbackDuration.
159 /// </summary>
160 void EndHit()
161 {
162 isHit = false;
163 }
164
165 /// <summary>
166 /// Destruye el enemigo al quedarse sin vida.
167 /// Usa un delay de 0.5 segundos para permitir efectos visuales de muerte
168 /// (animaciones, partículas, sonidos) antes de eliminar el objeto.
169 ///
170 /// CUÁNDO SE EJECUTA: Llamado desde TakeDamage() cuando currentHealth llega a 0.
171 /// </summary>
172 void Die()
173 {
174 Debug.Log("Enemigo derrotado");
175 // Delay para permitir efectos visuales antes de destruir
176 Destroy(gameObject, 0.5f);
177 }
178
179 // ============================================================
180 // FLUJO DEL SISTEMA
181 // ============================================================
182 /*
183 * 1. Un Hitbox o ExtendedHitbox detecta colisión con el enemigo.
184 * 2. Llaman a TakeDamage(damage, attackerPosition).
185 * 3. Se descuenta vida y se llama ApplyHit().
186 * 4. ApplyHit() verifica isHit → si false, aplica knockback y activa hit-stun.
187 * 5. Después de knockbackDuration segundos → EndHit() desactiva isHit.
188 * 6. Si currentHealth <= 0 → Die() destruye el GameObject con 0.5s de delay.
189 */
190
191 // ============================================================
192 // INTERACCIONES CON OTROS SISTEMAS
193 // ============================================================
194 /*
195 * ES LLAMADO POR:
196 * - Hitbox.cs → TakeDamage() cuando el hitbox de ataque toca al enemigo.
197 * - ExtendedHitbox.cs → TakeDamage() para ataques con hitbox extendido.
198 *
199 * AFECTA A:
200 * - Rigidbody2D del enemigo (knockback físico).
201 * - El GameObject completo (lo destruye al morir).
202 *
203 * NO AFECTA:
204 * - EnergySystem (la energía se gestiona en Hitbox.cs, no aquí).
205 * - UI de vida (el enemigo no tiene barra de vida en este sistema).
206 */
207
208 // ============================================================
209 // NOTAS / ADVERTENCIAS
210 // ============================================================
211 /*
212 * ⚠ Si el enemigo no tiene Rigidbody2D, el knockback no se aplica
213 * pero el daño y la muerte funcionan correctamente.
214 *
215 * ⚠ Die() tiene un delay de 0.5s. Durante ese tiempo el enemigo sigue
216 * presente en escena y puede recibir más golpes (aunque currentHealth = 0).
217 * Considerar agregar un flag isDead para ignorar daño adicional.
218 *
219 * ⚠ knockbackDuration NO debe ser negativo. Mathf.Max lo protege,
220 * pero configurar valores negativos en el Inspector genera comportamiento
221 * inesperado silencioso.
222 */
223}
