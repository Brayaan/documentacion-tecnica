1// ============================================================
2// ARCHIVO: ExtendedHitbox.cs
3// PROPÓSITO: Hitbox auxiliar que amplía el alcance de un ataque
4// reutilizando la configuración del Hitbox principal.
5// RESPONSABILIDAD: Detectar enemigos en un área extendida durante ataques activos
6// y aplicar daño con los mismos valores del Hitbox padre.
7// SISTEMA: Combate — Detección de Golpes (Hitbox Extendido)
8// ============================================================
9
10/*
11 * DESCRIPCIÓN DEL SISTEMA
12 * -----------------------
13 * ExtendedHitbox es un hitbox secundario que complementa al Hitbox principal.
14 * Comparte el valor de daño del Hitbox padre para garantizar consistencia,
15 * pero tiene su propio Collider2D con mayor alcance (configurado en la escena).
16 *
17 * A diferencia de Hitbox.cs (que usa OnTriggerStay2D para daño continuo),
18 * este script usa OnTriggerEnter2D para registrar el golpe solo al momento
19 * del primer contacto.
20 *
21 * Solo procesa golpes cuando PlayerAttack reporta un ataque activo,
22 * evitando detecciones falsas fuera de la animación de ataque.
23 *
24 * SISTEMA: Combate / Detección de Golpes
25 * INTERACTÚA CON: Hitbox.cs, PlayerAttack.cs, EnemyHealthSystem.cs, EnergySystem.cs
26 */
27
28using UnityEngine;
29
30public class ExtendedHitbox : MonoBehaviour
31{
32 // -------------------------
33 // VARIABLES PRIVADAS
34 // -------------------------
35
36 /// <summary>
37 /// Referencia al Hitbox principal del mismo GameObject.
38 /// Se usa para leer el valor de daño (originalHitbox.damage)
39 /// en lugar de duplicarlo, manteniendo un único punto de configuración.
40 /// </summary>
41 private Hitbox originalHitbox;
42
43 /// <summary>
44 /// Referencia al PlayerAttack en la raíz de la jerarquía del jugador.
45 /// Necesario para verificar si hay un ataque activo antes de procesar golpes.
46 /// </summary>
47 private PlayerAttack playerAttack;
48
49 // ============================================================
50 // EVENTOS DE UNITY
51 // ============================================================
52
53 /// <summary>
54 /// Start() — Se ejecuta una vez al activarse el objeto.
55 /// Obtiene referencias a Hitbox (mismo objeto) y PlayerAttack (raíz de jerarquía).
56 ///
57 /// NOTA: Si el objeto no es hijo del jugador, transform.root.GetComponent
58 /// puede devolver null y el hitbox quedará inactivo.
59 /// </summary>
60 void Start()
61 {
62 // Obtener el Hitbox principal del mismo GameObject
63 originalHitbox = GetComponent<Hitbox>();
64
65 // Obtener PlayerAttack desde la raíz de la jerarquía del jugador
66 playerAttack = transform.root.GetComponent<PlayerAttack>();
67 }
68
69 /// <summary>
70 /// OnTriggerEnter2D — Se ejecuta una vez al entrar en contacto con otro Collider2D.
71 /// Registra el impacto al PRIMER frame de contacto (no continuo como OnTriggerStay2D).
72 ///
73 /// CUÁNDO SE EJECUTA: Cuando el Collider2D de este hitbox toca otro collider.
74 /// QUÉ CONTROLA: Validaciones de ataque activo, aplicación de daño y ganancia de energía.
75 ///
76 /// FLUJO INTERNO:
77 /// 1. Verificar que las referencias críticas existen.
78 /// 2. Verificar que PlayerAttack está en estado de ataque activo.
79 /// 3. Verificar que el collider tocado tiene el tag "Enemy".
80 /// 4. Obtener EnemyHealthSystem y aplicar daño con el valor del Hitbox padre.
81 /// 5. Otorgar energía al atacante por conectar el golpe.
82 /// </summary>
83 void OnTriggerEnter2D(Collider2D other)
84 {
85 // Cancelar si faltan componentes críticos del sistema
86 if (originalHitbox == null || playerAttack == null)
87 return;
88
89 // Solo procesar durante un ataque activo y con el hitbox habilitado
90 if (!playerAttack.IsAttacking() || !gameObject.activeSelf)
91 return;
92
93 if (other.CompareTag("Enemy"))
94 {
95 EnemyHealthSystem enemyHealth = other.GetComponent<EnemyHealthSystem>();
96
97 if (enemyHealth != null)
98 {
99 // Posición del atacante para calcular la dirección del knockback en el enemigo
100 Vector2 attackerPosition = transform.root.position;
101
102 // Usar el daño del Hitbox padre para no duplicar la configuración
103 int damage = originalHitbox.damage;
104
105 enemyHealth.TakeDamage(damage, attackerPosition);
106
107 // Otorgar energía al atacante por conectar el golpe exitosamente
108 EnergySystem attackerEnergy = transform.root.GetComponent<EnergySystem>();
109 if (attackerEnergy != null)
110 {
111 // "Puñetazo" es el tipo de ataque fijo para el hitbox extendido
112 attackerEnergy.GainEnergyFromAttack("Puñetazo");
113 }
114
115 Debug.Log($"Golpe a enemigo!");
116 }
117 }
118 }
119
120 // ============================================================
121 // FLUJO DEL SISTEMA
122 // ============================================================
123 /*
124 * 1. PlayerAttack activa el hitbox extendido (SetActive(true)).
125 * 2. El Collider2D entra en contacto con un enemigo → OnTriggerEnter2D().
126 * 3. Se verifican: referencias, ataque activo, tag "Enemy".
127 * 4. EnemyHealthSystem.TakeDamage() se llama con el daño del Hitbox padre.
128 * 5. EnergySystem del atacante recibe +5 por "Puñetazo".
129 * 6. PlayerAttack desactiva el hitbox (SetActive(false)) al terminar el ataque.
130 */
131
132 // ============================================================
133 // INTERACCIONES CON OTROS SISTEMAS
134 // ============================================================
135 /*
136 * DEPENDE DE:
137 * - Hitbox.cs (mismo objeto) → para leer el valor de daño.
138 * - PlayerAttack.cs (raíz de jerarquía) → para verificar estado de ataque.
139 * - EnemyHealthSystem.cs (en el enemigo) → para aplicar el daño.
140 * - EnergySystem.cs (raíz de jerarquía) → para dar energía al atacante.
141 *
142 * ES ACTIVADO/DESACTIVADO POR:
143 * - PlayerAttack.cs → ActivarHitbox() / DesactivarHitbox()
144 *
145 * DIFERENCIA CON Hitbox.cs:
146 * - Hitbox.cs usa OnTriggerStay2D (daño continuo mientras hay contacto).
147 * - ExtendedHitbox usa OnTriggerEnter2D (daño único al primer contacto).
148 * - ExtendedHitbox solo afecta a "Enemy", no a "Player".
149 */
150
151 // ============================================================
152 // NOTAS / ADVERTENCIAS
153 // ============================================================
154 /*
155 * ⚠ El tipo de ataque para la energía está hardcodeado como "Puñetazo".
156 * Si este hitbox se reutiliza para patadas u otros ataques, la energía
157 * ganada siempre será la del puñetazo (+5). Considerar exponer
158 * attackName como variable pública para mayor flexibilidad.
159 *
160 * ⚠ No tiene cooldown propio (a diferencia de Hitbox.cs).
161 * Un enemigo puede recibir daño múltiple si múltiples hitboxes
162 * extendidos lo tocan en el mismo frame.
163 *
164 * ⚠ Si originalHitbox es null (porque el Hitbox.cs se eliminó),
165 * el script retorna silenciosamente sin daño. Verificar en el Inspector
166 * que ambos componentes estén en el mismo GameObject.
167 */
168}
