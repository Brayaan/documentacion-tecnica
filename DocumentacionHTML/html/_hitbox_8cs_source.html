1// ============================================================
2// ARCHIVO: Hitbox.cs
3// PROPÓSITO: Detectar colisiones de ataque y aplicar daño, knockback
4// y ganancia de energía a jugadores y enemigos.
5// RESPONSABILIDAD: Núcleo del sistema de daño por contacto en combate.
6// Maneja bloqueos, cooldown de golpe y comunicación
7// con los sistemas de salud y energía.
8// SISTEMA: Combate — Detección de Golpes (Hitbox Principal)
9// ============================================================
10
11/*
12 * DESCRIPCIÓN DEL SISTEMA
13 * -----------------------
14 * Hitbox es el componente que convierte un Collider2D en un área de daño activa.
15 * Usa OnTriggerStay2D para detectar contacto continuo mientras el Collider está activo,
16 * lo que permite que el golpe conecte aunque el personaje ya esté dentro del área.
17 *
18 * Implementa:
19 * - Cooldown por golpe (hitCooldown) para evitar daño excesivo por contacto prolongado.
20 * - Detección de bloqueo: si el objetivo bloquea, el atacante recibe un rebote.
21 * - Comunicación con HealthSystem, EnergySystem del atacante y del objetivo.
22 * - Guard contra autogolpe (ignora el propio personaje).
23 *
24 * Los hitboxes se activan/desactivan desde PlayerAttack.cs durante cada ataque.
25 *
26 * SISTEMA: Combate / Detección de Golpes
27 * INTERACTÚA CON: PlayerAttack.cs, PlayerDefense.cs, HealthSystem.cs, EnergySystem.cs
28 */
29
30using UnityEngine;
31
32public class Hitbox : MonoBehaviour
33{
34 // -------------------------
35 // VARIABLES PÚBLICAS
36 // -------------------------
37
38 /// <summary>
39 /// Nombre del ataque asociado a este hitbox (ej: "Puñetazo", "Patada").
40 /// Se usa para calcular la ganancia de energía en EnergySystem.GainEnergyFromAttack().
41 /// Configurable desde el Inspector por tipo de hitbox.
42 /// </summary>
43 public string attackName;
44
45 /// <summary>
46 /// Cantidad de daño que este hitbox inflige por golpe.
47 /// Configurable desde el Inspector. También lo lee ExtendedHitbox.cs.
48 /// </summary>
49 public int damage = 1;
50
51 /// <summary>
52 /// Tiempo mínimo en segundos entre golpes consecutivos mientras el Collider
53 /// permanece en contacto. Evita que un solo ataque quite vida múltiple
54 /// veces por frame durante el contacto.
55 /// </summary>
56 public float hitCooldown = 0.2f;
57
58 // -------------------------
59 // VARIABLES PRIVADAS
60 // -------------------------
61
62 /// <summary>
63 /// Referencia al PlayerAttack en la raíz de la jerarquía.
64 /// Necesaria para verificar si hay un ataque activo antes de aplicar daño.
65 /// </summary>
66 private PlayerAttack attack;
67
68 /// <summary>
69 /// Timestamp del último golpe registrado (Time.time en el momento del impacto).
70 /// Inicializado en float.NegativeInfinity para que el primer golpe siempre conecte.
71 /// Usado para calcular el cooldown entre golpes en OnTriggerStay2D.
72 /// </summary>
73 private float lastHitTime = float.NegativeInfinity;
74
75 // ============================================================
76 // EVENTOS DE UNITY
77 // ============================================================
78
79 /// <summary>
80 /// Start() — Se ejecuta una vez al activarse el hitbox en escena.
81 /// Obtiene la referencia a PlayerAttack desde la raíz de la jerarquía.
82 ///
83 /// NOTA: El hitbox generalmente es hijo del jugador, por lo que
84 /// transform.root apunta al GameObject raíz del jugador.
85 /// </summary>
86 void Start()
87 {
88 // Buscar PlayerAttack en la raíz de la jerarquía del jugador
89 attack = transform.root.GetComponent<PlayerAttack>();
90 }
91
92 /// <summary>
93 /// OnDisable() — Se ejecuta cuando el hitbox se desactiva (SetActive(false)).
94 /// Resetea el cooldown para que el próximo ataque siempre conecte
95 /// en el primer frame, independientemente del tiempo transcurrido.
96 ///
97 /// IMPORTANTE: Sin este reset, si el jugador ataca rápido, el cooldown
98 /// podría bloquear el golpe del siguiente ataque.
99 /// </summary>
100 void OnDisable()
101 {
102 // Resetear cooldown para que el próximo ataque siempre conecte
103 lastHitTime = float.NegativeInfinity;
104 }
105
106 /// <summary>
107 /// OnTriggerStay2D — Se ejecuta cada frame mientras otro Collider2D
108 /// permanece dentro del área del hitbox.
109 ///
110 /// CUÁNDO SE EJECUTA: Cada frame de contacto con un collider en trigger mode.
111 /// QUÉ CONTROLA: Validaciones de ataque activo, bloqueo del objetivo,
112 /// cooldown de golpe, aplicación de daño y energía.
113 ///
114 /// FLUJO INTERNO:
115 /// 1. Verificar que PlayerAttack existe y hay ataque activo.
116 /// 2. Verificar tag del objetivo ("Player" o "Enemy").
117 /// 3. Ignorar autogolpe (mismo root GameObject).
118 /// 4. Verificar si el objetivo está bloqueando → rebote al atacante.
119 /// 5. Verificar cooldown de golpe.
120 /// 6. Aplicar daño vía HealthSystem.
121 /// 7. Dar energía al objetivo (GainEnergyFromDamage) y al atacante (GainEnergyFromAttack).
122 /// </summary>
123 private void OnTriggerStay2D(Collider2D other)
124 {
125 // Cancelar si el ataque no está activo o el hitbox fue desactivado
126 if (attack == null || !attack.IsAttacking() || !gameObject.activeSelf)
127 return;
128
129 if (other.CompareTag("Player") || other.CompareTag("Enemy"))
130 {
131 // Evitar que el hitbox golpee al propio personaje que lo contiene
132 if (other.gameObject == transform.root.gameObject)
133 return;
134
135 PlayerDefense defense = other.GetComponent<PlayerDefense>();
136
137 // Si el objetivo está bloqueando: rebote al atacante (no al objetivo)
138 if (defense != null && defense.IsBlocking())
139 {
140 Debug.Log(other.name + " bloqueó el ataque");
141
142 Rigidbody2D attackerRb = transform.root.GetComponent<Rigidbody2D>();
143
144 if (attackerRb != null)
145 {
146 // Dirección del rebote: desde el bloqueador hacia el atacante
147 Vector2 direction = (transform.root.position - other.transform.position).normalized;
148
149 attackerRb.linearVelocity = Vector2.zero;
150 attackerRb.AddForce(new Vector2(direction.x * 4f, 1.5f), ForceMode2D.Impulse);
151 }
152
153 return; // Interrumpir: no aplicar daño al bloqueador
154 }
155
156 HealthSystem health = other.GetComponent<HealthSystem>();
157
158 if (health != null)
159 {
160 // Verificar cooldown: evitar golpes múltiples en contacto continuo
161 // Mathf.Max garantiza que hitCooldown negativo se trate como cero
162 if (Time.time - lastHitTime < Mathf.Max(0f, hitCooldown))
163 return;
164
165 lastHitTime = Time.time;
166
167 Vector2 attackerPosition = transform.root.position;
168
169 health.TakeDamage(damage, attackerPosition);
170
171 // Energía al objetivo: ganar por absorber el impacto
172 EnergySystem targetEnergy = other.GetComponent<EnergySystem>();
173 if (targetEnergy != null)
174 targetEnergy.GainEnergyFromDamage();
175
176 // Energía al atacante: ganar por conectar el golpe
177 EnergySystem attackerEnergy = transform.root.GetComponent<EnergySystem>();
178 if (attackerEnergy != null)
179 attackerEnergy.GainEnergyFromAttack(attackName);
180
181 Debug.Log("Golpeaste a: " + other.name + " con: " + attackName);
182 }
183 }
184 }
185
186 // ============================================================
187 // FLUJO DEL SISTEMA
188 // ============================================================
189 /*
190 * 1. PlayerAttack activa el hitbox (SetActive(true)) al inicio del ataque.
191 * 2. Si hay un Collider2D dentro del área → OnTriggerStay2D() cada frame.
192 * 3. Se validan: ataque activo, tag, autogolpe, bloqueo, cooldown.
193 * 4. Si pasa todas las validaciones → HealthSystem.TakeDamage() en el objetivo.
194 * 5. EnergySystem del objetivo recibe +2 (GainEnergyFromDamage).
195 * 6. EnergySystem del atacante recibe +5 o +10 según attackName.
196 * 7. PlayerAttack desactiva el hitbox → OnDisable() resetea lastHitTime.
197 */
198
199 // ============================================================
200 // INTERACCIONES CON OTROS SISTEMAS
201 // ============================================================
202 /*
203 * DEPENDE DE:
204 * - PlayerAttack.cs → IsAttacking() para validar estado del ataque.
205 * - PlayerDefense.cs → IsBlocking() para detectar bloqueos.
206 * - HealthSystem.cs → TakeDamage() para aplicar daño al objetivo.
207 * - EnergySystem.cs → GainEnergyFromDamage() y GainEnergyFromAttack().
208 *
209 * ES ACTIVADO/DESACTIVADO POR:
210 * - PlayerAttack.cs → ActivarHitbox() / DesactivarHitbox()
211 * → ActivarKickHitbox() / DesactivarKickHitbox()
212 *
213 * SU VALOR damage ES LEÍDO POR:
214 * - ExtendedHitbox.cs → originalHitbox.damage
215 *
216 * AFECTA A:
217 * - Rigidbody2D del atacante (rebote al bloquear).
218 * - HealthSystem del objetivo (daño).
219 * - EnergySystem del objetivo y del atacante.
220 */
221
222 // ============================================================
223 // NOTAS / ADVERTENCIAS
224 // ============================================================
225 /*
226 * ⚠ attackName DEBE coincidir exactamente con los strings en EnergySystem
227 * ("Puñetazo" o "Patada"). Un typo en el Inspector causará que
228 * la energía del atacante no aumente, sin ningún error visible.
229 *
230 * ⚠ OnTriggerStay2D se ejecuta en PhysicsUpdate, no en Update.
231 * Si el físico está a menor framerate, el cooldown puede comportarse
232 * de forma diferente al esperado a FPS muy bajos.
233 *
234 * ⚠ El rebote al bloquear tiene fuerza fija (4f, 1.5f) hardcodeada.
235 * Considerar exponer estas variables en el Inspector para balanceo.
236 *
237 * ⚠ Este script solo afecta a GameObjects con HealthSystem.
238 * Enemigos con EnemyHealthSystem NO son dañados por este Hitbox;
239 * para ellos se usa ExtendedHitbox.cs.
240 */
241}
