1// ============================================================
2// ARCHIVO: PlayerDefense.cs
3// PROPÓSITO: Gestionar el sistema de bloqueo del jugador:
4// input de bloqueo, animación, hitbox de guardia y ganancia de energía.
5// RESPONSABILIDAD: Detectar cuándo el jugador bloquea, activar el hitbox
6// de bloqueo, sincronizar la animación y dar energía al inicio.
7// SISTEMA: Combate — Defensa / Bloqueo
8// ============================================================
9
10/*
11 * DESCRIPCIÓN DEL SISTEMA
12 * -----------------------
13 * Este script controla la mecánica de defensa del jugador.
14 * Mientras se mantiene presionada la tecla de bloqueo (L por defecto),
15 * el jugador entra en estado de guardia:
16 * - Se activa el hitbox de bloqueo (que Hitbox.cs detecta para rebotar al atacante).
17 * - Se dispara la animación de bloqueo en el Animator.
18 * - Se otorga energía al inicio del bloqueo (solo una vez por activación).
19 *
20 * Al soltar la tecla, todo vuelve al estado normal.
21 * La lógica solo se ejecuta cuando el estado cambia (no cada frame si ya está bloqueando),
22 * lo que evita llamadas innecesarias a EnergySystem cada frame.
23 *
24 * SISTEMA: Combate / Defensa
25 * INTERACTÚA CON: Hitbox.cs, EnergySystem.cs, Animator
26 */
27
28using UnityEngine;
29
30public class PlayerDefense : MonoBehaviour
31{
32 // -------------------------
33 // REFERENCIAS EXTERNAS
34 // -------------------------
35
36 /// <summary>
37 /// GameObject del hitbox de bloqueo.
38 /// Cuando está activo, Hitbox.cs lo detecta mediante PlayerDefense.IsBlocking()
39 /// y aplica el rebote al atacante en lugar de dañar al bloqueador.
40 /// Debe asignarse desde el Inspector.
41 /// </summary>
42 public GameObject blockHitbox;
43
44 // -------------------------
45 // TECLAS DE INPUT
46 // -------------------------
47
48 /// <summary>
49 /// Tecla asignada al bloqueo. Por defecto: L.
50 /// Configurable desde el Inspector para remapeo de controles.
51 /// Usa GetKey (no GetKeyDown) para detectar pulsación continua.
52 /// </summary>
53 public KeyCode blockKey = KeyCode.L;
54
55 // -------------------------
56 // VARIABLES PRIVADAS
57 // -------------------------
58
59 /// <summary>
60 /// Estado actual del bloqueo.
61 /// true = el jugador está bloqueando activamente.
62 /// false = el jugador no está en guardia.
63 /// Este valor es el que Hitbox.cs consulta vía IsBlocking().
64 /// </summary>
65 private bool isBlocking = false;
66
67 /// <summary>
68 /// Referencia al Animator del jugador.
69 /// Se obtiene en Start(). Necesario para sincronizar la animación de bloqueo.
70 /// </summary>
71 private Animator anim;
72
73 // ============================================================
74 // EVENTOS DE UNITY
75 // ============================================================
76
77 /// <summary>
78 /// Start() — Se ejecuta una vez al iniciar.
79 /// Obtiene el Animator y desactiva el hitbox de bloqueo para asegurar
80 /// que el personaje comienza sin guardia activa.
81 /// </summary>
82 void Start()
83 {
84 anim = GetComponent<Animator>();
85
86 // Desactivar hitbox de bloqueo al iniciar para evitar guardia permanente
87 if (blockHitbox != null)
88 blockHitbox.SetActive(false);
89 }
90
91 /// <summary>
92 /// Update() — Se ejecuta cada frame.
93 /// Lee el input de bloqueo y compara con el estado anterior (isBlocking).
94 /// Solo ejecuta la lógica de cambio de estado cuando el input cambia,
95 /// evitando llamadas repetidas a EnergySystem y al Animator cada frame.
96 ///
97 /// CUÁNDO SE EJECUTA: Cada frame durante la partida.
98 /// QUÉ CONTROLA: Detección de cambio de estado de bloqueo,
99 /// ganancia de energía al inicio, animación y hitbox.
100 ///
101 /// FLUJO DE CAMBIO DE ESTADO:
102 /// Input presionado (isBlocking = false → true):
103 /// → EnergySystem.GainEnergyFromBlock() (solo una vez)
104 /// → Animator.SetBool("isBlocking", true)
105 /// → blockHitbox.SetActive(true)
106 /// Input soltado (isBlocking = true → false):
107 /// → Animator.SetBool("isBlocking", false)
108 /// → blockHitbox.SetActive(false)
109 /// </summary>
110 void Update()
111 {
112 bool input = Input.GetKey(blockKey);
113
114 // Solo ejecutar lógica cuando el estado de bloqueo cambia (no cada frame)
115 if (input != isBlocking)
116 {
117 isBlocking = input;
118
119 // Otorgar energía solo al inicio del bloqueo (transición false → true)
120 if (isBlocking)
121 {
122 EnergySystem energy = GetComponent<EnergySystem>();
123 if (energy != null)
124 energy.GainEnergyFromBlock();
125 }
126
127 // Sincronizar animación con el estado actual de bloqueo
128 if (anim != null)
129 anim.SetBool("isBlocking", isBlocking);
130
131 // Activar o desactivar el hitbox de bloqueo según el estado
132 if (blockHitbox != null)
133 blockHitbox.SetActive(isBlocking);
134 }
135 }
136
137 // ============================================================
138 // API PÚBLICA
139 // ============================================================
140
141 /// <summary>
142 /// Expone el estado de bloqueo para que Hitbox.cs pueda consultarlo
143 /// durante la detección de colisiones y decidir si aplicar daño
144 /// o hacer rebotar al atacante.
145 /// </summary>
146 /// <returns>true si el jugador está bloqueando; false si no.</returns>
147 public bool IsBlocking()
148 {
149 return isBlocking;
150 }
151
152 // ============================================================
153 // FLUJO DEL SISTEMA
154 // ============================================================
155 /*
156 * ACTIVAR BLOQUEO:
157 * 1. GetKey(L) = true + isBlocking = false → cambio de estado.
158 * 2. isBlocking = true.
159 * 3. EnergySystem.GainEnergyFromBlock() → +3 de energía (una sola vez).
160 * 4. Animator.SetBool("isBlocking", true) → animación de guardia.
161 * 5. blockHitbox.SetActive(true) → hitbox de bloqueo activo.
162 *
163 * MIENTRAS SE BLOQUEA:
164 * 6. input == isBlocking (ambos true) → no se ejecuta nada más.
165 * 7. Hitbox.cs detecta colisión → IsBlocking() = true → rebote al atacante.
166 *
167 * DESACTIVAR BLOQUEO:
168 * 8. GetKey(L) = false + isBlocking = true → cambio de estado.
169 * 9. isBlocking = false.
170 * 10. Animator.SetBool("isBlocking", false) → animación normal.
171 * 11. blockHitbox.SetActive(false) → hitbox desactivado.
172 */
173
174 // ============================================================
175 // INTERACCIONES CON OTROS SISTEMAS
176 // ============================================================
177 /*
178 * AFECTA A:
179 * - EnergySystem.cs → GainEnergyFromBlock() al inicio del bloqueo.
180 * - Animator → SetBool("isBlocking") para controlar animación.
181 * - blockHitbox (GameObject) → SetActive on/off.
182 *
183 * ES CONSULTADO POR:
184 * - Hitbox.cs → IsBlocking() para determinar si aplicar daño o rebote.
185 *
186 * NO DEPENDE DE ningún otro sistema de combate en runtime.
187 */
188
189 // ============================================================
190 // NOTAS / ADVERTENCIAS
191 // ============================================================
192 /*
193 * ⚠ La ganancia de energía (+3) ocurre SOLO al inicio del bloqueo
194 * (transición false → true), no mientras se mantiene bloqueando.
195 * Esto es intencional para no recompensar el bloqueo continuo.
196 *
197 * ⚠ El parámetro "isBlocking" del Animator debe existir como Bool
198 * en el Animator Controller. Si no existe, Unity lanzará un warning
199 * pero el sistema seguirá funcionando.
200 *
201 * ⚠ Este script no bloquea el input de ataque ni de movimiento mientras
202 * se está bloqueando. Si se desea que el bloqueo cancele ataques,
203 * añadir lógica en PlayerAttack.cs para verificar PlayerDefense.IsBlocking().
204 *
205 * ⚠ blockHitbox debe tener un Collider2D en modo Trigger para que
206 * Hitbox.cs pueda detectar el bloqueo correctamente mediante
207 * OnTriggerStay2D. Sin esto, el rebote al atacante no funcionará.
208 */
209}
