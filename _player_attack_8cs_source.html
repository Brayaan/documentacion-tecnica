1// ============================================================
2// ARCHIVO: PlayerAttack.cs
3// PROPÓSITO: Gestionar el sistema de ataques del jugador:
4// puñetazo y patada, sus hitboxes, cooldowns y animaciones.
5// RESPONSABILIDAD: Capturar input de ataque, activar hitboxes durante
6// la ventana de daño y sincronizar el estado de ataque
7// con PlayerMovement.
8// SISTEMA: Combate — Input y Control de Ataques
9// ============================================================
10
11/*
12 * DESCRIPCIÓN DEL SISTEMA
13 * -----------------------
14 * Este script es el controlador central de los ataques del jugador.
15 * Escucha el input de teclado para puñetazo (J) y patada (K),
16 * dispara las animaciones correspondientes, activa el hitbox durante
17 * la ventana de daño (attackDuration) y lo desactiva al terminar.
18 *
19 * Implementa:
20 * - Cooldown independiente para puñetazo y patada.
21 * - Flag isAttacking para evitar ataques superpuestos.
22 * - Sincronización del estado isAttacking con PlayerMovement.
23 * - API pública (IsAttacking()) para que Hitbox y ExtendedHitbox
24 * verifiquen si deben aplicar daño.
25 *
26 * SISTEMA: Combate / Input de Ataque
27 * INTERACTÚA CON: PlayerMovement.cs, Hitbox.cs, ExtendedHitbox.cs, Animator
28 */
29
30using UnityEngine;
31
32public class PlayerAttack : MonoBehaviour
33{
34 // -------------------------
35 // REFERENCIAS EXTERNAS
36 // -------------------------
37
38 /// <summary>
39 /// Referencia al script de movimiento del jugador.
40 /// Necesaria para sincronizar el flag isAttacking con PlayerMovement,
41 /// que puede usar ese dato para modificar el comportamiento del movimiento.
42 /// Debe asignarse desde el Inspector.
43 /// </summary>
44 public PlayerMovement playerMovement;
45
46 // -------------------------
47 // HITBOXES
48 // -------------------------
49
50 /// <summary>
51 /// GameObject del hitbox de puñetazo.
52 /// Se activa/desactiva durante la ventana de daño del puñetazo.
53 /// Debe asignarse desde el Inspector. El GameObject debe tener un
54 /// Collider2D en modo Trigger y el script Hitbox.cs.
55 /// </summary>
56 public GameObject punchHitbox;
57
58 /// <summary>
59 /// GameObject del hitbox de patada.
60 /// Se activa/desactiva durante la ventana de daño de la patada.
61 /// Debe asignarse desde el Inspector. El GameObject debe tener un
62 /// Collider2D en modo Trigger y el script Hitbox.cs.
63 /// </summary>
64 public GameObject kickHitbox;
65
66 // -------------------------
67 // TECLAS DE INPUT
68 // -------------------------
69
70 /// <summary>
71 /// Tecla asignada al puñetazo. Por defecto: J.
72 /// Configurable desde el Inspector para remapeo de controles.
73 /// </summary>
74 public KeyCode punchKey = KeyCode.J;
75
76 /// <summary>
77 /// Tecla asignada a la patada. Por defecto: K.
78 /// Configurable desde el Inspector para remapeo de controles.
79 /// </summary>
80 public KeyCode kickKey = KeyCode.K;
81
82 // -------------------------
83 // COOLDOWNS Y ESTADO
84 // -------------------------
85
86 /// <summary>
87 /// Flag que indica si la patada está disponible para ejecutarse.
88 /// Se pone en false al atacar y vuelve a true después de kickCooldown segundos.
89 /// </summary>
90 private bool kickReady = true;
91
92 /// <summary>
93 /// Tiempo en segundos que debe pasar antes de poder usar la patada de nuevo.
94 /// Configurable desde el Inspector.
95 /// </summary>
96 public float kickCooldown = 0.5f;
97
98 /// <summary>
99 /// Flag que indica si el puñetazo está disponible para ejecutarse.
100 /// Se pone en false al atacar y vuelve a true después de punchCooldown segundos.
101 /// </summary>
102 private bool punchReady = true;
103
104 /// <summary>
105 /// Tiempo en segundos que debe pasar antes de poder usar el puñetazo de nuevo.
106 /// Configurable desde el Inspector.
107 /// </summary>
108 public float punchCooldown = 0.3f;
109
110 /// <summary>
111 /// Flag global de ataque activo.
112 /// true = el jugador está ejecutando un ataque y no puede iniciar otro.
113 /// false = el jugador puede iniciar un nuevo ataque.
114 /// También se sincroniza con PlayerMovement.isAttacking.
115 /// </summary>
116 private bool isAttacking = false;
117
118 /// <summary>
119 /// Tiempo en segundos que el hitbox permanece activo durante un ataque.
120 /// Define la "ventana de daño": cuánto tiempo puede conectar el golpe.
121 /// También controla la duración del estado isAttacking.
122 /// </summary>
123 private float attackDuration = 0.3f;
124
125 // -------------------------
126 // REFERENCIAS INTERNAS
127 // -------------------------
128
129 /// <summary>
130 /// Referencia al Animator del jugador.
131 /// Se obtiene en Start(). Necesario para disparar triggers de animación.
132 /// </summary>
133 private Animator anim;
134
135 // ============================================================
136 // EVENTOS DE UNITY
137 // ============================================================
138
139 /// <summary>
140 /// Start() — Se ejecuta una vez al iniciar.
141 /// Obtiene el Animator y asegura que ambos hitboxes están desactivados.
142 /// Los hitboxes SIEMPRE deben comenzar desactivados para evitar
143 /// detecciones falsas al iniciar la escena.
144 /// </summary>
145 void Start()
146 {
147 anim = GetComponent<Animator>();
148
149 // Asegurar que los hitboxes están desactivados al iniciar
150 if (punchHitbox != null)
151 punchHitbox.SetActive(false);
152
153 if (kickHitbox != null)
154 kickHitbox.SetActive(false);
155 }
156
157 /// <summary>
158 /// Update() — Se ejecuta cada frame.
159 /// Lee el input de teclado para puñetazo y patada.
160 /// Verifica cooldown, flag de ataque activo y disponibilidad antes de atacar.
161 ///
162 /// CUÁNDO SE EJECUTA: Cada frame durante la partida.
163 /// QUÉ CONTROLA: Input de ataques, inicio de StartAttack(), activación de hitboxes,
164 /// triggers de animación y cooldowns.
165 ///
166 /// ORDEN DE VERIFICACIÓN (puñetazo):
167 /// 1. Input detectado (GetKeyDown).
168 /// 2. No hay ataque en curso (!isAttacking).
169 /// 3. Puñetazo disponible (punchReady).
170 /// → StartAttack() + animación + hitbox activo + cooldown.
171 /// </summary>
172 void Update()
173 {
174 // PUÑETAZO: verificar input, estado de ataque y cooldown
175 if (Input.GetKeyDown(punchKey) && !isAttacking && punchReady)
176 {
177 StartAttack();
178 anim.SetTrigger("punch");
179 ActivarHitbox();
180 Invoke(nameof(DesactivarHitbox), attackDuration);
181
182 punchReady = false;
183 Invoke(nameof(ResetPunch), punchCooldown);
184 }
185
186 // PATADA: verificar cooldown, input y estado de ataque
187 if (Input.GetKeyDown(kickKey) && kickReady && !isAttacking)
188 {
189 StartAttack();
190 anim.SetTrigger("kick");
191 ActivarKickHitbox();
192 Invoke(nameof(DesactivarKickHitbox), attackDuration);
193
194 kickReady = false;
195 Invoke(nameof(ResetKick), kickCooldown);
196 }
197 }
198
199 // ============================================================
200 // MÉTODOS PRINCIPALES
201 // ============================================================
202
203 /// <summary>
204 /// Inicia el estado de ataque global.
205 /// Pone isAttacking en true y lo sincroniza con PlayerMovement.
206 /// Programa EndAttack() para que se ejecute después de attackDuration.
207 ///
208 /// CUÁNDO SE LLAMA: Al inicio de cada ataque (puñetazo o patada).
209 /// QUÉ AFECTA: isAttacking, PlayerMovement.isAttacking.
210 /// </summary>
211 void StartAttack()
212 {
213 isAttacking = true;
214
215 // Sincronizar flag de ataque con PlayerMovement
216 if (playerMovement != null)
217 {
218 playerMovement.isAttacking = true;
219 }
220
221 Invoke(nameof(EndAttack), attackDuration);
222 }
223
224 /// <summary>
225 /// Finaliza el estado de ataque y libera el sistema para el próximo ataque.
226 /// Se ejecuta automáticamente mediante Invoke() al vencer attackDuration.
227 ///
228 /// CUÁNDO SE LLAMA: Después de attackDuration segundos desde StartAttack().
229 /// QUÉ AFECTA: isAttacking, PlayerMovement.isAttacking.
230 /// </summary>
231 void EndAttack()
232 {
233 isAttacking = false;
234
235 // Limpiar flag de ataque en PlayerMovement
236 if (playerMovement != null)
237 {
238 playerMovement.isAttacking = false;
239 }
240 }

247 /// Expone el estado de ataque para que Hitbox y ExtendedHitbox
248 /// puedan verificar si deben procesar colisiones.
249 /// </summary>
250 /// <returns>true si hay un ataque en curso; false en caso contrario.</returns>
251 public bool IsAttacking()
252 {
253 return isAttacking;
254 }

261 public void ActivarHitbox()
262 {
263 if (punchHitbox != null)
264 punchHitbox.SetActive(true);
265 }
266
267 /// <summary>Desactiva el hitbox de puñetazo. Llamado al terminar la ventana de daño.</summary>
268 public void DesactivarHitbox()
269 {
270 if (punchHitbox != null)
271 punchHitbox.SetActive(false);
272 }
273
274 /// <summary>Activa el hitbox de patada. Llamado al inicio de la patada.</summary>
275 public void ActivarKickHitbox()
276 {
277 if (kickHitbox != null)
278 kickHitbox.SetActive(true);
279 }
280
281 /// <summary>Desactiva el hitbox de patada. Llamado al terminar la ventana de daño.</summary>
282 public void DesactivarKickHitbox()
283 {
284 if (kickHitbox != null)
285 kickHitbox.SetActive(false);
286 }
287
288 // ============================================================
289 // MÉTODOS AUXILIARES — RESET DE COOLDOWNS
290 // ============================================================
291
292 /// <summary>
293 /// Reactiva la disponibilidad de la patada después de kickCooldown segundos.
294 /// Se ejecuta mediante Invoke() desde Update().
295 /// </summary>
296 void ResetKick()
297 {
298 kickReady = true;
299 }
300
301 /// <summary>
302 /// Reactiva la disponibilidad del puñetazo después de punchCooldown segundos.
303 /// Se ejecuta mediante Invoke() desde Update().
304 /// </summary>
305 void ResetPunch()
306 {
307 punchReady = true;
308 }
309
310 // ============================================================
311 // FLUJO DEL SISTEMA
312 // ============================================================
313 /*
314 * PUÑETAZO:
315 * 1. GetKeyDown(J) + !isAttacking + punchReady → StartAttack().
316 * 2. anim.SetTrigger("punch") → inicia animación.
317 * 3. punchHitbox.SetActive(true) → ventana de daño abierta.
318 * 4. Después de attackDuration → DesactivarHitbox() + EndAttack().
319 * 5. Después de punchCooldown → ResetPunch() → punchReady = true.
320 *
321 * PATADA: mismo flujo con kickHitbox, trigger "kick" y kickCooldown.
322 */
323
324 // ============================================================
325 // INTERACCIONES CON OTROS SISTEMAS
326 // ============================================================
327 /*
328 * AFECTA A:
329 * - PlayerMovement.isAttacking → sincronización de estado.
330 * - Hitbox.cs → lee IsAttacking() para validar golpes.
331 * - ExtendedHitbox.cs → lee IsAttacking() para validar golpes.
332 * - Animator → triggers "punch" y "kick".
333 * - punchHitbox / kickHitbox → SetActive on/off.
334 *
335 * NO DEPENDE DE ningún sistema de combate en runtime
336 * (solo necesita PlayerMovement, Animator e Inspector).
337 */
338
339 // ============================================================
340 // NOTAS / ADVERTENCIAS
341 // ============================================================
342 /*
343 * ⚠ attackDuration controla tanto la ventana de daño del hitbox
344 * como la duración del estado isAttacking. Si se necesita que
345 * la ventana de daño sea diferente a la duración del estado de ataque,
346 * separar en dos variables distintas.
347 *
348 * ⚠ Invoke() no se cancela si el script se deshabilita durante el ataque.
349 * Si PlayerAttack se deshabilita a mitad del ataque, EndAttack() y
350 * los reset de hitboxes podrían no ejecutarse, dejando el sistema en
351 * estado inválido. Considerar usar CancelInvoke() en OnDisable().
352 *
353 * ⚠ Los triggers de Animator "punch" y "kick" deben existir en el
354 * Animator Controller. Si no existen, Unity lanzará un warning
355 * y la animación no se reproducirá, pero el resto del sistema seguirá
356 * funcionando normalmente.
357 */
358}
