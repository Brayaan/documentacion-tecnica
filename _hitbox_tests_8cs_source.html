108 /// <summary>
109 /// CASO 1:
110 /// El hitbox no debe aplicar daño múltiples veces dentro del cooldown.
111 /// </summary>
112 [UnityTest]
113 public IEnumerator Hitbox_DoesNotDamageMultipleTimesWithinCooldown()
114 {
115 Hitbox hitbox = CreateAttacker();
116
117 LogAssert.Expect(LogType.Error, "healthImage no está asignada en el Inspector");
118 HealthSystem health = CreateTarget();
119 yield return null;
120
121 PlayerAttack playerAttack = _attackerGO.GetComponent<PlayerAttack>();
122 _isAttackingField.SetValue(playerAttack, true);
123
124 Collider2D targetCol = _targetGO.GetComponent<Collider2D>();
125 int initialHealth = health.currentHealth;
126
127 // Primer impacto válido
128 LogAssert.Expect(LogType.Error, "healthImage no está asignada en el Inspector");
129 _onTriggerStay2D.Invoke(hitbox, new object[] { targetCol });
130
131 int healthAfterFirstHit = health.currentHealth;
132
133 // Segundo intento dentro del cooldown (no debe aplicar daño)
134 _onTriggerStay2D.Invoke(hitbox, new object[] { targetCol });
135
136 Assert.AreEqual(initialHealth - hitbox.damage, healthAfterFirstHit);
137 Assert.AreEqual(healthAfterFirstHit, health.currentHealth);
138 }
