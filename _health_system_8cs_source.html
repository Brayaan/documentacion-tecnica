1// ============================================================
2// ARCHIVO: HealthSystem.cs
3// PROPÓSITO: Gestionar la vida del jugador, el daño recibido,
4// el knockback, la animación de hit y la UI de salud.
5// RESPONSABILIDAD: Punto de entrada de daño para el jugador.
6// Coordina física, animación e input durante el hit-stun.
7// SISTEMA: Combate — Salud del Jugador
8// ============================================================
9
10/*
11 * DESCRIPCIÓN DEL SISTEMA
12 * -----------------------
13 * Este script controla toda la lógica de salud del jugador.
14 * Cuando un enemigo (o el hitbox de otro jugador) golpea al personaje,
15 * llama a TakeDamage() para:
16 * 1. Reducir la vida.
17 * 2. Actualizar la UI (spritesheet de corazones).
18 * 3. Aplicar knockback físico.
19 * 4. Deshabilitar el input del jugador durante el hit-stun.
20 * 5. Activar la animación de recibir golpe.
21 *
22 * Implementa un sistema de hit-stun (isHit) que bloquea golpes apilados
23 * y restaura el control del jugador al expirar el timer.
24 *
25 * SISTEMA: Combate / Salud
26 * INTERACTÚA CON: Hitbox.cs, PlayerMovement.cs, Animator
27 */
28
29using UnityEngine;
30using UnityEngine.UI;
31
32public class HealthSystem : MonoBehaviour
33{
34 // -------------------------
35 // VARIABLES PÚBLICAS
36 // -------------------------
37
38 /// <summary>
39 /// Vida máxima del jugador. Define el techo de salud posible.
40 /// Configurable desde el Inspector. No debe ser 0 o negativo.
41 /// </summary>
42 public int maxHealth = 100;
43
44 /// <summary>
45 /// Vida actual del jugador en tiempo real.
46 /// Se inicializa igual a maxHealth en Start().
47 /// Visible en Inspector para debug durante la partida.
48 /// </summary>
49 public int currentHealth;
50
51 /// <summary>
52 /// Componente Image de Unity UI que muestra visualmente la barra de vida.
53 /// Debe asignarse desde el Inspector. Si es null, la UI no se actualizará.
54 /// </summary>
55 public Image healthImage;
56
57 /// <summary>
58 /// Fuerza del empuje físico al recibir un golpe.
59 /// Configurable desde el Inspector. Valor más alto = mayor desplazamiento.
60 /// </summary>
61 public float knockbackForce = 5f;
62
63 /// <summary>
64 /// Duración en segundos del hit-stun:
65 /// tiempo mínimo antes de poder recibir otro knockback y tiempo
66 /// que el input del jugador permanece deshabilitado.
67 /// Recomendación: valor bajo (0.05f) para respuesta fluida,
68 /// mayor si se desea un hit-stun más pronunciado.
69 /// </summary>
70 public float knockbackDuration = 0.05f;
71
72 // -------------------------
73 // VARIABLES PRIVADAS
74 // -------------------------
75
76 /// <summary>
77 /// Referencia al Rigidbody2D del jugador.
78 /// Necesario para aplicar el knockback físico. Se obtiene en Start().
79 /// </summary>
80 private Rigidbody2D rb;
81
82 /// <summary>
83 /// Referencia al script PlayerMovement del jugador.
84 /// Se deshabilita durante el hit-stun para bloquear el input.
85 /// Se restaura al finalizar el hit-stun.
86 /// </summary>
87 private PlayerMovement movement;
88
89 /// <summary>
90 /// Referencia al Animator del jugador.
91 /// Se usa para disparar el trigger "Hit" si el Animator lo tiene registrado.
92 /// </summary>
93 private Animator anim;
94
95 /// <summary>
96 /// Flag de hit-stun activo.
97 /// true = el jugador está en hit-stun, no puede moverse ni recibir otro knockback.
98 /// false = el jugador puede recibir daño con efecto físico completo.
99 /// </summary>
100 private bool isHit = false;
101
102 /// <summary>
103 /// Array de sprites cargados desde el spritesheet de corazones.
104 /// Cargado automáticamente desde Resources/HeartCounter/heart_counter-Sheet.
105 /// Cada elemento corresponde a un nivel de vida diferente.
106 /// </summary>
107 private Sprite[] healthSprites;
108
109 // ============================================================
110 // EVENTOS DE UNITY
111 // ============================================================
112
113 /// <summary>
114 /// Start() — Se ejecuta una vez al iniciar la escena.
115 /// Carga el spritesheet de vida, inicializa currentHealth al máximo,
116 /// obtiene referencias a Rigidbody2D, PlayerMovement y Animator,
117 /// y actualiza la UI al estado inicial.
118 /// </summary>
119 void Start()
120 {
121 // Cargar spritesheet de vida desde la carpeta Resources
122 healthSprites = Resources.LoadAll<Sprite>("HeartCounter/heart_counter-Sheet");
123
124 currentHealth = maxHealth;
125
126 rb = GetComponent<Rigidbody2D>();
127 movement = GetComponent<PlayerMovement>();
128 anim = GetComponent<Animator>();
129
130 UpdateHealthUI();
131 }
132
133 // ============================================================
134 // MÉTODOS PRINCIPALES
135 // ============================================================
136
137 /// <summary>
138 /// Punto de entrada de daño para el jugador.
139 /// Reduce la vida, actualiza la UI y aplica el hit-stun con knockback.
140 ///
141 /// CUÁNDO SE LLAMA: Desde Hitbox.cs cuando el hitbox de un enemigo o jugador
142 /// toca el collider del jugador.
143 /// QUÉ AFECTA: currentHealth, UI de vida, física del Rigidbody2D,
144 /// input del jugador (PlayerMovement), Animator.
145 /// </summary>
146 /// <param name="damage">Cantidad de puntos de vida a restar.</param>
147 /// <param name="attackerPosition">Posición del atacante para calcular dirección del knockback.</param>
148 public void TakeDamage(int damage, Vector2 attackerPosition)
149 {
150 currentHealth -= damage;
151
152 // Clampear vida para no bajar de cero
153 if (currentHealth < 0)
154 currentHealth = 0;
155
156 Debug.Log("Vida actual: " + currentHealth);
157
158 UpdateHealthUI();
159
160 ApplyHit(attackerPosition);
161 }
162
163 // ============================================================
164 // MÉTODOS AUXILIARES
165 // ============================================================
166
167 /// <summary>
168 /// Aplica el hit-stun al jugador: deshabilita input, activa animación
169 /// de golpe y aplica knockback físico.
170 ///
171 /// CUÁNDO SE EJECUTA: Llamado internamente desde TakeDamage().
172 /// QUÉ AFECTA: PlayerMovement (deshabilitado), Animator (trigger "Hit"),
173 /// Rigidbody2D (velocidad y fuerza), flag isHit.
174 ///
175 /// NOTA: Si isHit ya está activo, el método retorna sin hacer nada.
176 /// Esto es intencional para evitar stacking de knockback.
177 /// </summary>
178 /// <param name="attackerPosition">Posición desde donde vino el ataque.</param>
179 void ApplyHit(Vector2 attackerPosition)
180 {
181 // Descartar golpe si el hit-stun ya está activo
182 if (isHit) return;
183
184 isHit = true;
185
186 // Deshabilitar input del jugador durante el hit-stun
187 if (movement != null)
188 movement.enabled = false;
189
190 // Activar animación de recibir golpe si existe el parámetro en el Animator
191 if (anim != null && TieneParametro("Hit"))
192 anim.SetTrigger("Hit");
193
194 // Calcular dirección del empuje opuesta al atacante
195 Vector2 direction = (transform.position - (Vector3)attackerPosition).normalized;
196
197 if (rb != null)
198 {
199 rb.linearVelocity = Vector2.zero;
200 // El eje Y tiene un valor fijo de 2f para dar pequeña elevación al impacto
201 rb.AddForce(new Vector2(direction.x * knockbackForce, 2f), ForceMode2D.Impulse);
202 }
203
204 // Mathf.Max evita que un knockbackDuration negativo rompa el Invoke
205 Invoke(nameof(EndHit), Mathf.Max(0f, knockbackDuration));
206 }
207
208 /// <summary>
209 /// Verifica si el Animator tiene un parámetro registrado por nombre.
210 /// Evita errores al llamar SetTrigger con un nombre que no existe.
211 ///
212 /// USO EXCLUSIVO: Solo llamado desde ApplyHit() para validar el parámetro "Hit".
213 /// </summary>
214 /// <param name="nombre">Nombre del parámetro a buscar en el Animator.</param>
215 /// <returns>true si el parámetro existe; false si no.</returns>
216 bool TieneParametro(string nombre)
217 {
218 foreach (var param in anim.parameters)
219 if (param.name == nombre) return true;
220 return false;
221 }
222
223 /// <summary>
224 /// Finaliza el hit-stun: reactiva el input del jugador y restablece isHit.
225 /// Se llama automáticamente mediante Invoke() al vencer knockbackDuration.
226 /// </summary>
227 void EndHit()
228 {
229 isHit = false;
230
231 // Restaurar input del jugador al salir del hit-stun
232 if (movement != null)
233 movement.enabled = true;
234 }
235
236 /// <summary>
237 /// Actualiza el sprite de la barra de vida en la UI según currentHealth.
238 /// Calcula el índice proporcional en el spritesheet e invierte el orden
239 /// para que el sprite 0 del array corresponda a vida llena.
240 ///
241 /// CUÁNDO SE EJECUTA: Llamado desde Start() y después de cada TakeDamage().
242 ///
243 /// LÓGICA DE INVERSIÓN:
244 /// index normal = proporcional a la vida actual (0 = vacío, max = lleno)
245 /// index invertido = (Length-1) - index → el array comienza desde lleno
246 /// </summary>
247 void UpdateHealthUI()
248 {
249 // Guard: evitar división por cero si maxHealth es inválido
250 if (maxHealth <= 0)
251 {
252 Debug.LogError("maxHealth debe ser mayor que cero", this);
253 return;
254 }
255
256 if (healthImage == null)
257 {
258 Debug.LogError("healthImage no está asignada en el Inspector", this);
259 return;
260 }
261
262 if (healthSprites == null || healthSprites.Length == 0)
263 {
264 Debug.LogError("No se cargaron los sprites", this);
265 return;
266 }
267
268 // Calcular índice proporcional según la vida actual
269 int index = Mathf.RoundToInt(((float)currentHealth / maxHealth) * (healthSprites.Length - 1));
270
271 // Invertir: index 0 del array = barra llena
272 index = (healthSprites.Length - 1) - index;
273
274 healthImage.sprite = healthSprites[index];
275 }
276
277 // ============================================================
278 // FLUJO DEL SISTEMA
279 // ============================================================
280 /*
281 * 1. Start() → carga sprites, inicializa vida al máximo, actualiza UI.
282 * 2. Hitbox.cs detecta colisión → llama TakeDamage(damage, attackerPosition).
283 * 3. TakeDamage() → descuenta vida, actualiza UI, llama ApplyHit().
284 * 4. ApplyHit() → verifica isHit, deshabilita PlayerMovement,
285 * activa animación "Hit", aplica fuerza al Rigidbody2D.
286 * 5. Después de knockbackDuration → EndHit() → reactiva PlayerMovement.
287 * 6. Si currentHealth = 0, el jugador está muerto
288 * (no hay lógica de muerte aquí; extender si se requiere).
289 */
290
291 // ============================================================
292 // INTERACCIONES CON OTROS SISTEMAS
293 // ============================================================
294 /*
295 * ES LLAMADO POR:
296 * - Hitbox.cs → TakeDamage() cuando el hitbox de un atacante toca al jugador.
297 *
298 * AFECTA A:
299 * - PlayerMovement.cs → se deshabilita/habilita durante el hit-stun.
300 * - Animator → trigger "Hit" para reproducir animación de golpe.
301 * - Rigidbody2D → aplicación de knockback.
302 * - healthImage (UI) → actualización visual de la barra de vida.
303 *
304 * NO GESTIONA:
305 * - Muerte del jugador (no hay lógica de Game Over aquí).
306 * - EnergySystem (la energía al recibir daño se gestiona en Hitbox.cs).
307 */
308
309 // ============================================================
310 // NOTAS / ADVERTENCIAS
311 // ============================================================
312 /*
313 * ⚠ El spritesheet DEBE estar en Resources/HeartCounter/heart_counter-Sheet.
314 * Si la ruta o el nombre cambian, los sprites no se cargarán.
315 *
316 * ⚠ No hay lógica de muerte del jugador en este script.
317 * Si currentHealth llega a 0, nada ocurre más allá de la UI.
318 * Implementar Game Over o respawn externamente.
319 *
320 * ⚠ TieneParametro() itera todos los parámetros del Animator cada vez.
321 * En Animators muy grandes podría tener impacto mínimo. Si se necesita
322 * optimizar, cachear el resultado en Start().
323 *
324 * ⚠ knockbackDuration NO debe ser negativo. Mathf.Max lo protege,
325 * pero valores negativos en Inspector generan comportamiento inesperado.
326 */
327}
